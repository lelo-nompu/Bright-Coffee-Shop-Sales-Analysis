--- Understanding how much data we have and how diverse the items are to determine if it is suitable for analysis ---
SELECT
       COUNT(*) AS total_rows,
       COUNT(DISTINCT transaction_id) AS total_sales,
       COUNT(DISTINCT product_id) AS total_products,
       COUNT(DISTINCT store_id) AS total_stores
       FROM workspace.default.bright_coffee_shop_sales;

--- Analyse transaction by day of the week to identify which days drive the most sales---

SELECT 
       Dayname(transaction_date) AS day_name,
       COUNT(DISTINCT transaction_id) AS number_of_sales,
       SUM(transaction_qty * unit_price) AS revenue
       FROM workspace.default.bright_coffee_shop_sales
       GROUP BY day_name
       ORDER BY revenue DESC;

--- Analyse transaction by month of the year to identify broader trends and patterns over time ---

SELECT 
       Monthname(transaction_date) AS month_name,
       COUNT(DISTINCT transaction_id) AS number_of_sales,
       SUM(transaction_qty * unit_price) AS revenue
       FROM workspace.default.bright_coffee_shop_sales
       GROUP BY month_name
       ORDER BY revenue DESC;

--- Calculate the total revenue generated per transaction ---

SELECT transaction_id, 
       product_type,
       transaction_qty,
       unit_price,
       (transaction_qty * unit_price) as revenue_per_transaction
FROM workspace.default.bright_coffee_shop_sales;

--- Total number of units sols accross all transactions ---

SELECT SUM(transaction_qty) AS total_units_sold
FROM workspace.default.bright_coffee_shop_sales;

--- Calculate the total revnue generated ---

SELECT 
  SUM(transaction_qty * unit_price) AS total_revenue
FROM workspace.default.bright_coffee_shop_sales;

--- Determine which products generate the most revenue by summing total sales for each product type ---

SELECT product_type,
       SUM(transaction_qty * unit_price) AS total_revenue
FROM workspace.default.bright_coffee_shop_sales
GROUP BY product_type
ORDER BY total_revenue DESC;

--- Analyse product demand by calculating the total number of units sold per product type ---

SELECT 
       product_type,
       SUM(transaction_qty) AS total_units_sold
FROM workspace.default.bright_coffee_shop_sales
GROUP BY product_type
ORDER BY total_units_sold DESC;

--- Analyse sales performance across different times of the day by grouping transactions into time categories ---

SELECT
       CASE 
       WHEN date_format(transaction_time, 'HH:mm:ss') BETWEEN '06:00:00' AND '11:59:59' THEN 'Morning'
       WHEN date_format(transaction_time, 'HH:mm:ss') BETWEEN '12:00:00' AND '16:59:59' THEN 'Afternoon'
       WHEN date_format(transaction_time, 'HH:mm:ss') BETWEEN '17:00:00' AND '20:59:59' THEN 'Evening'
       ELSE 'Night'
  END AS time_category,
  SUM(transaction_qty * unit_price) AS revenue
FROM workspace.default.bright_coffee_shop_sales
GROUP BY time_category
ORDER BY revenue DESC;

--- Calculate average sales to understand typical revenue performance across different time periods ---

SELECT 
       Dayname(transaction_date) AS day_name,
       AVG(transaction_qty * unit_price) AS avg_sales
FROM workspace.default.bright_coffee_shop_sales
GROUP BY day_name;

--- Analyse sales trends over time by calculating total revenue per day ---

SELECT 
  transaction_date,
  SUM(transaction_qty * unit_price) AS daily_revenue
FROM workspace.default.bright_coffee_shop_sales
GROUP BY transaction_date
ORDER BY transaction_date;

--- Compare revenue accross the different store locations to identify which location performs best ---

SELECT 
       store_location,
       SUM(transaction_qty * unit_price) AS revenue
FROM workspace.default.bright_coffee_shop_sales
GROUP BY store_location
ORDER BY revenue DESC;

--- Classify products as high vs low performing based on their total revenue contribution ---
SELECT
       product_type,
       SUM(transaction_qty * unit_price) AS revenue,
       CASE 
               WHEN SUM(transaction_qty * unit_price) > 5000 THEN 'High Performing'
               ELSE 'Low Performing'
               END AS performance
FROM workspace.default.bright_coffee_shop_sales
GROUP BY product_type
ORDER BY revenue DESC;
